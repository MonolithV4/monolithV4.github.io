function deepMergeSeed(saved){
  const out = structuredClone(seed);
  if(!saved) return out;
  Object.assign(out, saved);
  out.settings = {...seed.settings,...(saved.settings||{})};
  out.schedule = saved.schedule || seed.schedule;
  out.habits = saved.habits || seed.habits;
  out.habitLogs = saved.habitLogs || {};
  out.activity = saved.activity || {};
  out.foodLogs = saved.foodLogs || [];
  out.customFoods = saved.customFoods || [];
  out.macroGoals = saved.macroGoals || seed.macroGoals;
  out.sleepLogs = saved.sleepLogs || [];
  out.measurements = saved.measurements || [];
  out.workouts = saved.workouts || [];
  out.plateInventory = saved.plateInventory || seed.plateInventory;
  return out;
}

function unitToGrams(qty,unit,food){
  qty=Number(qty||0);
  const serving=Number(document.getElementById('foodServingOverride')?.value||food?.serving||100);
  if(unit==='oz')return qty*28.3495;
  if(unit==='lb')return qty*453.59237;
  if(unit==='kg')return qty*1000;
  if(unit==='serving')return qty*serving;
  return qty;
}
